function AB_grotha()
{
var ax_grot = parseFloat(document.getElementById('ax').value);
var ay_grot = parseFloat(document.getElementById('ay').value);
var bx_grot = parseFloat(document.getElementById('bx').value);
var by_grot = parseFloat(document.getElementById('by').value);

var potega = Math.pow(ax_grot-bx_grot,2);
var potega1 = Math.pow(ay_grot-by_grot,2);
var dlugosc = Math.sqrt(potega+potega1);
document.getElementById("DLUGOSC").innerHTML = "Bok AB = "+dlugosc;
}
