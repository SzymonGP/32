    function drabina_grotha(x_grot,d_grot)
{
var x_grot = parseFloat(document.getElementById('dlugosc1').value);
var d_grot = parseFloat(document.getElementById('dlugosc2').value);
var drabina_grot = Math.sqrt(x_grot*x_grot+d_grot*d_grot);
 document.getElementsByName('drabina')[0].value=drabina_grot.toFixed(2);
}

	function alfa_grotha(x_grot,d_grot)
{
var x_grot = parseFloat(document.getElementById('dlugosc1').value);
var d_grot = parseFloat(document.getElementById('dlugosc2').value);
var a_grot = x_grot/d_grot;
var rad_grot = Math.atan(a_grot);
var alfa_grot = (rad_grot*180)/Math.PI;

  document.getElementsByName('alfa')[0].value=alfa_grot.toFixed(2);

}
